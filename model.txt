users {
  id int pk
  email string unique
  password string
  fullname string
  role enum('admin','customer')
  created_at timestamp
  updated_at timestamp
}

categories {
  id int pk
  name string unique
  slug string unique
  created_at timestamp
}

products {
  id int pk
  title string
  slug string unique
  description text
  price decimal
  stock int
  sku string
  category_id int fk
  is_active boolean
  created_at timestamp
  updated_at timestamp
}

product_images {
  id int pk
  product_id int fk
  url string
  is_primary boolean
  created_at timestamp
}

carts {
  id int pk
  user_id int fk unique
  created_at timestamp
}

cart_items {
  id int pk
  cart_id int fk
  product_id int fk
  quantity int
  created_at timestamp
}

orders {
  id int pk
  user_id int fk
  total_amount decimal
  status enum('pending','delivered','canceled')
  address text
  payment_method string
  created_at timestamp
  updated_at timestamp
}

order_items {
  id int pk
  order_id int fk
  product_id int fk
  quantity int
  price_at_purchase decimal
}

wishlists {
  id int pk
  user_id int fk
  product_id int fk
  created_at timestamp
}

reviews {
  id int pk
  user_id int fk
  product_id int fk
  rating tinyint
  title string
  comment text
  created_at timestamp
}

products.category_id > categories.id
product_images.product_id > products.id
carts.user_id > users.id
cart_items.cart_id > carts.id
cart_items.product_id > products.id
orders.user_id > users.id
order_items.order_id > orders.id
order_items.product_id > products.id
wishlists.user_id > users.id
wishlists.product_id > products.id
reviews.user_id > users.id
reviews.product_id > products.id
